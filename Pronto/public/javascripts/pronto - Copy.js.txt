var cart = [];
var tmp = [];

//fonction pour montrer l'onglet selectionné
function showBox(n) {

    hide('menu');
    hide('entrees');
    hide('plats');
    hide('desserts');
    hide('boissons');
    hide('commande');

    document.getElementById(n).style.display='block';


}

//fonction qui cache les autres onglets
function hide(id) {
    document.getElementById(id).style.display='none';
}
